CREATE TABLE ACCTYPE (
	ACCTYPE_ID INT(11) NOT NULL AUTO_INCREMENT,
	ACCTYPE_NAME VARCHAR(25) NOT NULL,
	ACCTYPE_DESC VARCHAR(300) NOT NULL,
	ACCTYPE_STAT BIT(1) NOT NULL DEFAULT 0,
	PRIMARY KEY (ACCTYPE_ID)
)

CREATE TABLE ACC (
	ACCID INT(11) NOT NULL AUTO_INCREMENT,
	ACCTYPE_FK INT(11) NOT NULL,
	ACCUSERNAME VARCHAR(25) NOT NULL,
	ACCPASSWORD VARCHAR(25) NOT NULL,
	PRIMARY KEY (ACCID),
	FOREIGN KEY (ACCTYPE_FK) REFERENCES ACCTYPE (ACCTYPE_ID),
	CREATED_AT TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
	ACC_STAT BIT(1) NOT NULL DEFAULT 0

)

CREATE TABLE PROGRAM
(
	PROG_ID INT(11) NOT NULL AUTO_INCREMENT,
	PROG_NAME VARCHAR(50) NOT NULL,
	PROG_INIT VARCHAR(10) NOT NULL,
	PRIMARY KEY (PROG_ID),
	PROG_STAT BIT(1) NOT NULL DEFAULT 0

)

CREATE TABLE ACADYR (
	ACADYR INT(11) NOT NULL AUTO_INCREMENT,
	ACADYR_NAME VARCHAR(50) NOT NULL,
	ACADYR_STAT BIT(1) NOT NULL DEFAULT 0,
	PRIMARY KEY (ACADYR)
)

CREATE TABLE BUILDING (
	BLDG_ID INT(11) NOT NULL AUTO_INCREMENT,
	BLDG_NAME VARCHAR(50) NOT NULL,
	BLDG_STAT BIT(1) NOT NULL DEFAULT 0,
	PRIMARY KEY (BLDG_ID)
)

CREATE TABLE ROOM (
	ROOM_ID INT(11) NOT NULL AUTO_INCREMENT,
	ROOM_NAME VARCHAR(50) NOT NULL,
	BLDG_FK INT(11) NOT NULL,
	ROOM_STAT BIT(1) NOT NULL DEFAULT 0,
	PRIMARY KEY (ROOM_ID),
	FOREIGN KEY (BLDG_ID) REFERENCES BUILDING(BLDG_ID)

)

CREATE TABLE SCHED (
	SCHED_ID INT(11) NOT NULL AUTO_INCREMENT,
	ACADYR_FK INT(11) NOT NULL,
	PROG_FK INT(11) NOT NULL,
	ROOM_FK INT(11) NOT NULL,
	SCHED_TIME VARCHAR(50) NOT NULL,
	SCHED_DATE DATE NOT NULL,
	SCHED_STAT BIT(1) NOT NULL DEFAULT 0,
	FOREIGN KEY (PROG_FK) REFERENCES PROGRAM(PROG_ID),
	FOREIGN KEY (ACADYR_FK) REFERENCES ACADYR(ACADYR_ID),
	FOREIGN KEY (ROOM_FK) REFERENCES ROOM(ROOM_ID),
	PRIMARY KEY (SCHED_ID),
	--SCHEDDTL_FK INT(11) NOT NULL,
	--FOREIGN KEY (SCHEDDTL_FK) REFERENCES SCHED_DETAILS(SCHEDDTL_ID)

)

CREATE TABLE APPLICANT (
	APP_ID INT(11) NOT NULL AUTO_INCREMENT,
	PRIMARY KEY (APP_ID),
	--APP_NUM VARCHAR(15) NOT NULL,
	--UNIQUE (APP_NUM),
	APP_FNAME VARCHAR(50) NOT NULL,
	APP_LNAME VARCHAR(50) NOT NULL,
	APP_MNAME VARCHAR(50),
	APP_SEX BIT(1) NOT NULL,
	APP_BDATE DATE NOT NULL,
	APP_PBIRTH VARCHAR(100) NOT NULL,
	APP_CIVIL VARCHAR(25) NOT NULL,
	APP_CITIZENSHIP VARCHAR(25) NOT NULL,
	APP_DISABILITY VARCHAR(100) NOT NULL DEFAULT "NONE",
	APP_PIC MEDIUMBLOB,
	APP_CONTACT VARCHAR(25) NOT NULL,
	APP_EMAIL VARCHAR(50) NOT NULL,
	APP_FIRST_FK INT(11) NOT NULL,
	APP_SECOND_FK INT(11),
	APP_THIRD_FK INT(11),
	SCHED_FK INT(11) NOT NULL,
	MOTHERNAME VARCHAR(100),
	MOTHERCIV VARCHAR(25),
	MOTHEROCCU VARCHAR(50),
	MOTHEREMP VARCHAR(100),
	FATHERNAME VARCHAR(100),
	FATHERCIV VARCHAR(25),
	FATHEROCCU VARCHAR(50),
	FATHEREMP VARCHAR(100),
	GUARDNAME VARCHAR(100),
	GUARDCIV VARCHAR(25),
	GUARDOCCU VARCHAR(50),
	GUARDEMP VARCHAR(100),
	ACC_FK INT(11) NOT NULL,
	FOREIGN KEY (ACC_FK) REFERENCES ACC (ACCID),
	FOREIGN KEY (APP_FIRST_FK) REFERENCES PROGRAM (PROG_ID),
	FOREIGN KEY (APP_SECOND_FK) REFERENCES PROGRAM (PROG_ID),
	FOREIGN KEY (APP_THIRD_FK) REFERENCES PROGRAM (PROG_ID),
	FOREIGN KEY (SCHED_FK) REFERENCES SCHED (SCHED_ID)

	)
